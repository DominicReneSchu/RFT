	\section{Introduction and Objectives}
	%% ============================================================
	
	Resonance Field Theory (RFT) is an axiomatic framework
	that describes oscillatory coupling as a universal organizing
	principle of physical and systemic phenomena.
	The goal is a parametric extension of the Planck relation
	that uniformly captures resonance phenomena across scales
	and domains -- from particle physics through cosmology
	to financial markets and nuclear technology.
	
	This document summarizes the theoretical foundations, the
	methodology, and the empirical results. The
	complete manuscript in IOP format is available as
	\texttt{rft\_manuscript\_en\_iop.tex}.
	All derivations, source codes, and data are openly
	accessible at
	\url{https://github.com/DominicReneSchu/RFT}.
	
	Philosophical and metaphysical extensions of the RFT
	are documented in a separate document
	(\texttt{RFT\_Philosophy.tex}) and explicitly marked as
	speculative perspectives.

	\section{Theoretical Background}
	%% ============================================================
	
	The RFT formulates seven hierarchically structured axioms
	(A1--A7), from which the central
	\textit{Resonance Field Equation} follows:
	\begin{equation}
		E = \pi \cdot \varepsilon(\Delta\varphi) \cdot \hbar \cdot f
	\end{equation}
	Here:
	\begin{itemize}
		\item $\pi$ is the mathematical constant as a measure of cyclic structure,
		\item $\hbar$ is the reduced Planck action quantum,
		\item $f$ is the resonance frequency of the system,
		\item $\varepsilon(\Delta\varphi) \in [0,1]$ is the
		phase-dependent \textbf{coupling operator}, which describes the
		strength of the resonance coupling between two systems
		as a function of their phase difference $\Delta\varphi$.
	\end{itemize}
	
	\subsection{Conceptual Distinctions}
	
	\begin{itemize}
		\item \textbf{Coupling operator}
		$\varepsilon(\Delta\varphi)$:
		Dimensionless parameter in the basic formula.
		Simplest model:
		$\varepsilon = \frac{1}{2}(1 + \cos\Delta\varphi)
		= \cos^2(\Delta\varphi/2)$.
		Limit values: $\varepsilon = 0$ (anti-resonance),
		$\varepsilon = 1$ (complete phase coherence).
		\item \textbf{Coupling efficiency}
		$\eta(\Delta\varphi)$:
		Measurable quantity in the FLRW simulations. Extracted from the
		time-averaged cross-term of two coupled scalar fields.
		Analytical expectation:
		$\eta_{\mathrm{theo}} = \cos^2(\Delta\varphi/2)$.
		\item \textbf{Identity $\varepsilon = \eta$}:
		In the resonance reactor context it was shown that
		$\varepsilon(\Delta\varphi) = \eta(\Delta\varphi)$
		holds exactly.
		This identity eliminates the coupling strength parameter
		$\kappa$ completely:
		$\kappa = 1$ (no free parameter).
	\end{itemize}
	
	\subsection{Planck Special Case}
	
	For $\varepsilon = 1/(2\pi)$ it follows that
	$E = \frac{1}{2}\hbar f$, consistent with the
	ground-state energy of the harmonic oscillator.
	
	\subsection{The Seven Axioms}
	
	\begin{enumerate}
		\item[A1] \textbf{Universal Oscillation:}
		Every physical entity can be described by an
		oscillation function.
		\item[A2] \textbf{Superposition and Interference:}
		Oscillations superpose linearly.
		\item[A3] \textbf{Resonance Condition:}
		Coupling occurs when frequencies stand in rational
		ratios ($f_1/f_2 = n/m$).
		\item[A4] \textbf{Coupling Energy:}
		$E = \pi \cdot \varepsilon(\Delta\varphi)
		\cdot \hbar \cdot f$.
		\item[A5] \textbf{Energy Direction:}
		Energy is a vectorial quantity with magnitude
		and direction in the resonance field.
		\item[A6] \textbf{Information Flow:}
		Information propagates exclusively along
		coherent resonance paths.
		\item[A7] \textbf{Invariance:}
		Resonance structures are preserved under variation of
		the observer's perspective and system parameters.
	\end{enumerate}

	\section{Methodology and Software Architecture}
	%% ============================================================
	
	The empirical validation is carried out by four independent
	simulation pipelines:
	
	\subsection{Monte Carlo Resonance Analysis}
	
	\begin{itemize}
		\item $50\,000$ pseudo-experiments on CMS open data
		($99\,915$ dielectron events)
		\item KDE background model with narrow signal exclusion
		\item Binomial test with Bonferroni correction over
		68~window widths
		\item Bootstrap confidence intervals
		($10\,000$~repetitions)
		\item Systematics checks over three KDE bandwidths
		and ten independent seeds
		\item \textbf{Total scope: $1\,500\,000$ simulations}
	\end{itemize}
	
	\subsection{Coupled FLRW Simulations}
	
	\begin{itemize}
		\item $1\,530$~individual simulations
		(51~$H_0$ points $\times$ 30~phase values)
		\item Klein-Gordon equation in expanding spacetime
		\item DOP853 integrator, $t = 0$--$120$,
		$12\,000$~evaluation points
		\item Jackknife error on $d_\eta$,
		bootstrap error on slope
	\end{itemize}
	
	\subsection{Resonance Reactor Simulation}
	
	\begin{itemize}

		\item Coupling strength $\kappa = 1$ from
		$\varepsilon = \eta$ identity
		\item Energy balance with Q-factor analysis
	\end{itemize}
	
	\subsection{ResoTrade (Application Concept)}
	
	\begin{itemize}

		of RFT axioms in a real market environment
		\item AC/DC decomposition (A1), superposition
